$$
\lim_{x \to 0} (1 + x)^{\frac{1}{x}}
$$

$$
\lim_{x \to 0} (1 + |x|)^{\frac{1}{x^{2}}}
$$


<div id="limits-plot" style="width:100%;max-width:800px;height:450px;"></div>

<!-- Plotly 불러오기 -->
<script src="https://cdn.plot.ly/plotly-2.35.2.min.js"></script>

<script>
document.addEventListener('DOMContentLoaded', function () {
  var plotDiv = document.getElementById('limits-plot');
  if (!plotDiv) {
    console.error('limits-plot div not found');
    return;
  }

  // 혹시 CDN이 막혀서 Plotly가 안 뜨는 경우를 체크
  if (!window.Plotly) {
    plotDiv.innerHTML = 'Plotly가 로드되지 않았습니다. (CDN 차단 여부를 확인해 보세요)';
    console.error('Plotly is not defined');
    return;
  }

  // (1 + x)^{1/x} 데이터
  var x1 = [];
  var y1 = [];
  for (var t = -0.9; t <= 0.3; t += 0.01) {
    if (Math.abs(t) < 1e-6) continue;
    if (1 + t <= 0) continue;
    x1.push(t);
    y1.push(Math.pow(1 + t, 1 / t));
  }

  // (1 + |x|)^{1/x^2} 데이터
  var x2 = [];
  var y2 = [];
  for (var s = -0.3; s <= 0.3; s += 0.005) {
    if (Math.abs(s) < 1e-6) continue;
    var base = 1 + Math.abs(s);
    var val = Math.pow(base, 1 / (s * s));
    if (val > 50) continue; // 발산 부분 잘라내기
    x2.push(s);
    y2.push(val);
  }

  var trace1 = {
    x: x1,
    y: y1,
    mode: 'lines',
    name: '(1 + x)^{1/x}'
  };

  var trace2 = {
    x: x2,
    y: y2,
    mode: 'lines',
    name: '(1 + |x|)^{1/x^2}'
  };

  var layout = {
    title: '함수 (1 + x)^{1/x} 와 (1 + |x|)^{1/x^2} 의 그래프 (x = 0 근처)',
    xaxis: { title: 'x' },
    yaxis: { title: 'y', rangemode: 'tozero' }
  };

  Plotly.newPlot('limits-plot', [trace1, trace2], layout);
});
</script>
