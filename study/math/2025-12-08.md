\[
\lim_{x \to 0} (1 + x)^{\frac{1}{x}}
\]

\[
\lim_{x \to 0} (1 + |x|)^{\frac{1}{x^{2}}}
\]

<div id="limits-plot" style="width:100%;max-width:800px;height:450px;"></div>

<!-- Plotly 불러오기 -->
<script src="https://cdn.plot.ly/plotly-latest.min.js"></script>

<script>
// (1 + x)^{1/x} 를 그리기 위한 데이터
var x1 = [];
var y1 = [];
for (var t = -0.9; t <= 0.3; t += 0.01) {
  if (Math.abs(t) < 1e-6) continue;        // 0 근처 제거
  if (1 + t <= 0) continue;                // 밑이 음수가 되지 않도록
  x1.push(t);
  y1.push(Math.pow(1 + t, 1 / t));
}

// (1 + |x|)^{1/x^2} 를 그리기 위한 데이터
var x2 = [];
var y2 = [];
for (var s = -0.3; s <= 0.3; s += 0.005) {
  if (Math.abs(s) < 1e-6) continue;        // 0 근처 제거
  var base = 1 + Math.abs(s);
  var val = Math.pow(base, 1 / (s * s));
  if (val > 50) continue;                  // 너무 크게 발산하는 부분은 잘라냄 (시각화용)
  x2.push(s);
  y2.push(val);
}

// 두 그래프 trace 정의
var trace1 = {
  x: x1,
  y: y1,
  mode: 'lines',
  name: '(1 + x)^{1/x}'
};

var trace2 = {
  x: x2,
  y: y2,
  mode: 'lines',
  name: '(1 + |x|)^{1/x^2}'
};

// 레이아웃 설정
var layout = {
  title: '함수 (1 + x)^{1/x} 와 (1 + |x|)^{1/x^2} 의 그래프 (x = 0 근처)',
  xaxis: { title: 'x' },
  yaxis: { title: 'y', rangemode: 'tozero' }
};

// 그래프 그리기
Plotly.newPlot('limits-plot', [trace1, trace2], layout);
</script>
